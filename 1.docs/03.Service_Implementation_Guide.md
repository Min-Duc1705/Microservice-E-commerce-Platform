### 2. `Program.cs`

Đăng ký Services, Database (EF Core) và MassTransit.

```csharp
using Microsoft.EntityFrameworkCore;
using MassTransit;
using PaymentService.Data;
using PaymentService.Consumers;

var builder = WebApplication.CreateBuilder(args);

// ======= 1. Đăng ký Database =======
builder.Services.AddDbContext<PaymentDbContext>(options =>
    options.UseNpgsql(builder.Configuration.GetConnectionString("DefaultConnection")));

// ======= 2. Đăng ký MassTransit =======
builder.Services.AddMassTransit(x =>
{
    // Đăng ký Consumer sẽ chạy ngầm
    x.AddConsumer<OrderCreatedConsumer>();

    x.UsingRabbitMq((context, cfg) =>
    {
        cfg.Host(builder.Configuration["RabbitMQ:Host"], "/", h =>
        {
            h.Username(builder.Configuration["RabbitMQ:Username"]);
            h.Password(builder.Configuration["RabbitMQ:Password"]);
        });

        // Tự động map Consumer vào Queue
        cfg.ConfigureEndpoints(context);
    });
});

builder.Services.AddControllers();
var app = builder.Build();
app.MapControllers();
app.Run();
```

### 3. `Consumers/OrderCreatedConsumer.cs`

Khi có đơn hàng mới (từ OrderService), tạo 1 bản ghi `PaymentTransaction` trạng thái `Pending`.

```csharp
using MassTransit;
using PaymentService.Data;
using PaymentService.Models;
using PaymentService.Utils.Enum;
// Lưu ý: OrderCreatedEvent cần được định nghĩa trong dự án CommonService (class class library gộp chung)
using CommonService.Events;

namespace PaymentService.Consumers;

public class OrderCreatedConsumer : IConsumer<OrderCreatedEvent>
{
    private readonly PaymentDbContext _dbContext;

    public OrderCreatedConsumer(PaymentDbContext dbContext)
    {
        _dbContext = dbContext;
    }

    public async Task Consume(ConsumeContext<OrderCreatedEvent> context)
    {
        var message = context.Message;

        var transaction = new PaymentTransaction
        {
            OrderId = message.OrderId,
            CustomerId = message.CustomerId,
            Amount = message.TotalAmount,
            Method = message.PaymentMethod,
            Status = PaymentStatus.Pending,
            CreatedAt = DateTime.UtcNow
        };

        _dbContext.PaymentTransactions.Add(transaction);
        await _dbContext.SaveChangesAsync();
    }
}
```

### 4. `Controllers/PaymentController.cs`

Xử lý tạo link VNPay và hứng Webhook từ VNPay.

```csharp
using Microsoft.AspNetCore.Mvc;
using MassTransit;
using PaymentService.Data;
using CommonService.Events;

namespace PaymentService.Controllers;

[ApiController]
[Route("api/payments")]
public class PaymentController : ControllerBase
{
    private readonly PaymentDbContext _dbContext;
    private readonly IPublishEndpoint _publishEndpoint;

    public PaymentController(PaymentDbContext dbContext, IPublishEndpoint publishEndpoint)
    {
        _dbContext = dbContext;
        _publishEndpoint = publishEndpoint;
    }

    // 1. Webhook hứng từ VNPay chuyển về
    [HttpGet("vnpay-return")]
    public async Task<IActionResult> VnPayReturn([FromQuery] string vnp_TxnRef, [FromQuery] string vnp_ResponseCode)
    {
        // Thực tế cần kiểm tra vnp_SecureHash bằng thư viện của VNPay để chống giả mạo.
        // Ở đây giả lập logic tìm OrderId từ vnp_TxnRef (OrderId được map vào TxnRef lúc tạo link).

        if (Guid.TryParse(vnp_TxnRef, out Guid orderId))
        {
            var transaction = _dbContext.PaymentTransactions.FirstOrDefault(x => x.OrderId == orderId);

            if (transaction != null && transaction.Status == Utils.Enum.PaymentStatus.Pending)
            {
                if (vnp_ResponseCode == "00") // 00 là thành công của VNPay
                {
                    transaction.Status = Utils.Enum.PaymentStatus.Success;
                    transaction.PaidAt = DateTime.UtcNow;

                    // Publish Event báo cho OrderService là đã thu tiền thành công
                    await _publishEndpoint.Publish(new PaymentCompletedEvent
                    {
                        OrderId = orderId,
                        Status = "Success"
                    });
                }
                else
                {
                    transaction.Status = Utils.Enum.PaymentStatus.Failed;
                    transaction.FailureReason = $"VNPay Error Code: {vnp_ResponseCode}";

                    await _publishEndpoint.Publish(new PaymentCompletedEvent
                    {
                        OrderId = orderId,
                        Status = "Failed"
                    });
                }
                await _dbContext.SaveChangesAsync();
            }
        }

        // Chuyển hướng người dùng về trang Frontend (Angular)
        return Redirect("http://localhost:4200/order-result?success=" + (vnp_ResponseCode == "00"));
    }
}
```

### 2. `Consumers/OrderCompletedConsumer.cs`

Hứng Event từ OrderService (khi đơn hàng chuyển trạng thái Completed) để tính Doanh Thu.

```csharp
using MassTransit;
using Microsoft.EntityFrameworkCore;
using ReportService.Data;
using ReportService.Models;
using ReportService.Utils.Enum;
using CommonService.Events; // Định nghĩa Event

namespace ReportService.Consumers;

public class OrderCompletedConsumer : IConsumer<OrderCompletedEvent>
{
    private readonly ReportDbContext _dbContext;

    public OrderCompletedConsumer(ReportDbContext dbContext)
    {
        _dbContext = dbContext;
    }

    public async Task Consume(ConsumeContext<OrderCompletedEvent> context)
    {
        var msg = context.Message;
        var dateNow = DateTime.UtcNow;
        var dailyLabel = dateNow.ToString("yyyy-MM-dd");

        // Gom tổng doanh thu & chi phí từ các item trong Event
        decimal totalRev = 0;
        decimal totalCost = 0;
        int itemsSold = 0;

        foreach(var item in msg.Items)
        {
            totalRev += item.Quantity * item.UnitPrice;
            totalCost += item.Quantity * item.UnitCost;
            itemsSold += item.Quantity;
        }

        // Tìm hoặc Tạo mới báo cáo NGÀY (Daily)
        var report = await _dbContext.RevenueReports
            .FirstOrDefaultAsync(x => x.Period == ReportPeriod.Daily && x.PeriodLabel == dailyLabel);

        if (report == null)
        {
            report = new RevenueReport
            {
                Period = ReportPeriod.Daily,
                PeriodLabel = dailyLabel,
                TotalRevenue = totalRev,
                TotalCost = totalCost,
                TotalOrders = 1,
                TotalItemsSold = itemsSold
            };
            _dbContext.RevenueReports.Add(report);
        }
        else
        {
            report.TotalRevenue += totalRev;
            report.TotalCost += totalCost;
            report.TotalOrders += 1;
            report.TotalItemsSold += itemsSold;
            report.LastUpdatedAt = dateNow;
        }

        await _dbContext.SaveChangesAsync();
    }
}
```

### 3. `Controllers/ReportController.cs`

Bây giờ việc viết API xem báo cáo doanh thu siêu dễ vì dữ liệu đã cộng gộp sẵn!

```csharp
using Microsoft.AspNetCore.Mvc;
using Microsoft.EntityFrameworkCore;
using ReportService.Data;
using ReportService.Utils.Enum;

namespace ReportService.Controllers;

[ApiController]
[Route("api/reports")]
public class ReportController : ControllerBase
{
    private readonly ReportDbContext _dbContext;

    public ReportController(ReportDbContext dbContext)
    {
        _dbContext = dbContext;
    }

    [HttpGet("revenue/daily")]
    public async Task<IActionResult> GetDailyRevenue([FromQuery] int days = 7)
    {
        // Lấy 7 ngày gần nhất, order theo ngày tăng dần để vẽ biểu đồ line chart
        var data = await _dbContext.RevenueReports
            .Where(x => x.Period == ReportPeriod.Daily)
            .OrderByDescending(x => x.PeriodLabel)
            .Take(days)
            .ToListAsync();

        return Ok(data.OrderBy(x => x.PeriodLabel));
    }
}
```
